    def MSLeNet(x,kp):
        # paranoia; reshape input to our needs
        x = tf.reshape(x, shape=[-1, 32, 32, 1])
        # first layer:
        #   - convolutional layer
        #   - 3x3 kernel
        #   - valid padding
        #   - 8 output channels
        ##   - with bias and ReLU activation
        #       32x32x1   -->   30x30x8
        W1 = tf.Variable(tf.random_normal([3, 3, 1, 8],stddev=0.1),name='W1')
        b1 = tf.Variable(tf.zeros([8]),name='b1')
        x = tf.nn.conv2d(x, W1, strides=[1,1,1,1], padding='VALID')
        x = tf.nn.bias_add(x,b1)
        x = tf.nn.relu(x)
        # second layer:
        #   - pooling layer
        #   - 2x2 kernel
        #       30x30x8   -->   15x15x8
        x = tf.nn.max_pool(x,ksize=[1,2,2,1],strides=[1,2,2,1],padding='VALID')
        # third layer:
        #   - convolutional layer
        #   - 6x6 kernel
        #   - valid padding
        #   - 21 output channels
        ##   - with bias and ReLU activation
        #       15x15x8   -->   10x10x21
        W3 = tf.Variable(tf.random_normal([6, 6, 8, 21],stddev=0.1),name='W3')
        b3 = tf.Variable(tf.zeros([21]),name='b3')
        x = tf.nn.conv2d(x, W3, strides=[1,1,1,1], padding='VALID')
        x = tf.nn.bias_add(x,b3)
        x = tf.nn.relu(x)
        # 4th layer:
        #   - pooling layer
        #   - 2x2 kernel
        #       10x10x21   -->   5x5x21
